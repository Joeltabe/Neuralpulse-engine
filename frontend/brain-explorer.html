# Load pretrained TRIBE v2 from HuggingFace and predict brain responses
from tribev2 import TribeModel

# 1. Load the model (downloads weights on first run)
model = TribeModel.from_pretrained(
    "facebook/tribev2",
    cache_folder="./cache",
    device="cpu",             # or "cuda" for GPU
)

# 2. Build events from your input
#    — text is auto-converted to speech and transcribed
df = model.get_events_dataframe(
    video_path="path/to/video.mp4"   # or text_path / audio_path
)

# 3. Run inference — returns (preds, segments)
#    preds:   (n_timesteps, 20484) numpy array of cortical vertex activations
#    segments: aligned time segments with metadata
preds, segments = model.predict(events=df)

print(preds.shape)  # e.g. (12, 20484)

# 4. Average the time dimension before ROI aggregation
vertex_scores = preds.mean(axis=0)  # length ~= 20,484 fsaverage5 cortical vertices

# 5. Map vertices to your ROI atlas, then POST JSON to this UI
roi_payload = {
    "scores": {"attention": 0.0, "dopamine": 0.0, "memory": 0.0},
    "regions": [
        {"name": "V1/V2 (Primary Visual)", "value": 74, "impact": "attention"},
        {"name": "FFA (Fusiform Face Area)", "value": 91, "impact": "attention"},
        {"name": "Amygdala", "value": 84, "impact": "dopamine"},
    ],
}

# Frontend contract: brainViewer.loadActivations(roi_payload)
